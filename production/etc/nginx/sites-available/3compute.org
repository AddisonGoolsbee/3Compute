# Redirect HTTP -> HTTPS for both domains
server {
    listen 80;
    server_name www.3compute.org;
    return 301 https://$host$request_uri;
}

server {
    listen 80;
    server_name api.3compute.org;
    return 301 https://$host$request_uri;
}

# HTTPS server for the frontend
server {
    listen 443 ssl;
    server_name www.3compute.org;

    ssl_certificate /etc/letsencrypt/live/www.3compute.org/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/www.3compute.org/privkey.pem;
    include /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem;

    # Serve static frontend files
    root /var/www/3compute/frontend/build/client;
    index index.html;

    location / {
        try_files $uri $uri/ /index.html;
    }
}

# HTTPS server for the backend API
server {
    listen 443 ssl;
    server_name api.3compute.org;

    ssl_certificate /etc/letsencrypt/live/api.3compute.org/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/api.3compute.org/privkey.pem;
    include /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem;

    location / {
        proxy_pass http://127.0.0.1:5555;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }
}
